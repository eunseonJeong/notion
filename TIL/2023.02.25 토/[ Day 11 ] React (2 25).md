    ```jsx
    import React, { useState } from 'react'
    
    function App() {
      const [number,setnumber] = useState(0);
    
      const isPlus = () =>{
        setnumber(number+1)       //**number에 값을 변경해주고 싶다면 setnumber만 건들여라!**
      };
      const isminus = () =>{
        setnumber(number-1)
      };
    
      return (
        <div>
          <div>{number}</div>
          <button onClick={isPlus}>플러스</button>
          <button onClick={isminus}>마이너스</button>
        </div>
      )
    }
    
    export default App
    ```
